import csv import re def extrair_links(texto): """Extrai links específicos do texto do post.""" links = { 'dood': '', 'lulu': '', 'vidnest': '' } # Regex para encontrar URLs urls = re.findall(r'https?://[^\s]+', texto) for url in urls: if 'doodstream.com' in url or 'dood.li' in url: links['dood'] = url elif 'lulustream.com' in url or 'lulu.to' in url: links['lulu'] = url elif 'vidnest.io' in url: links['vidnest'] = url return links def processar_dados(input_file, output_file): with open(input_file, mode='r', encoding='utf-8') as infile: # Pula a primeira linha se for o cabeçalho original reader = csv.DictReader(infile) fieldnames = ['THUMB', 'NEW', 'NOME', 'DOOD', 'LULU', 'VIDNEST', 'SITE', 'TITLE', 'ID', 'TAGS', 'POST', 'VIEWS'] with open(output_file, mode='w', encoding='utf-8', newline='') as outfile: writer = csv.DictWriter(outfile, fieldnames=fieldnames) writer.writeheader() for row in reader: post_content = row.get('POST', '') links = extrair_links(post_content) # Extração básica de ID e Nome (ajuste conforme a necessidade) nome_match = re.search(r'\{New\}\s+([^\s]+ [^\s]+)', post_content) nome = nome_match.group(1) if nome_match else "" # Tentativa de pegar o ID numérico que costuma aparecer nos links ou títulos id_match = re.search(r'/(\d+)', post_content) id_val = id_match.group(1) if id_match else "" new_row = { 'THUMB': row.get('THUMB', ''), 'NEW': '{New}', 'NOME': nome, 'DOOD': links['dood'], # Coluna D 'LULU': links['lulu'], # Coluna E 'VIDNEST': links['vidnest'], # Coluna F 'SITE': 'Onlyfans', # Valor padrão ou extraído 'TITLE': '(2026)', 'ID': id_val, 'TAGS': '', # Pode ser extraído dos #hashtags se necessário 'POST': post_content, 'VIEWS': row.get('VIEWS', '0') } writer.writerow(new_row) # Exemplo de uso: # processar_dados('Restart.xlsx - Página1.csv', 'Planilha_Corrigida.csv') print("Script pronto para processar e restaurar as colunas D, E e F.")
